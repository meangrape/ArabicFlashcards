%% Commands for managing arabic
% polyglossia doesn't support LuaTex, hence
% the need for thise
% \textarabic inserts arabic into a L2R langauge
% \n handles numbers inside of arabic text
% \afootnote creates footnotes in arabic
% \begin{Arabic} \end{Arabic} marks an area
% where all the text is in arabic

\newfontfamily\arabicfont
[Path             = ./fonts/pragmata/regular/,
 Extension        = .ttf,
 Script           = Arabic,
 Scale            = 4.0
]{PragmataRegular}

%% Set up Arabic 
\newcommand{\textarabic}[1]     % Arabic inside LTR
    {\bgroup\textdir TRT\arabicfont #1\egroup}
\newcommand{\n}         [1]     % for digits inside Arabic text
    {\bgroup\textdir TLT #1\egroup}
\newcommand{\afootnote} [1]     % Arabic footnotes
    {\footnote{\textarabic{#1}}}
\newenvironment{Arabic}     % Arabic paragraph
    {\textdir TRT\pardir TRT\arabicfont}{}
